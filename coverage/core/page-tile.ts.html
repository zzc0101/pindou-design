/**
 * 大图分页打印（A3/A4）
 *
 * @author zhuzc
 * @date 2026-06-22
 *
 * 当 BeadGrid 超过单页容量（默认 35×50 = 1750 格），自动分成多页打印。
 * 每页带拼接标记（边缘标页码 + 拼接线），方便用户拼接施工。
 */
 
import type { BeadGrid } from '@/types/bead';
import { EMPTY_CELL } from '@/core/color-quantize';
import type { RenderCommand } from './bead-renderer';
 
/** 分页选项 */
export interface TileOptions {
  /** 每页格子数 X（默认 35，适合 A4 横向留拼接边距） */
  cellsPerPageX: number;
  /** 每页格子数 Y（默认 50） */
  cellsPerPageY: number;
  /** 拼接重叠格数（默认 2，方便拼接对齐） */
  overlap: number;
  /** 背景色 */
  background: string;
  /** 透明背景 */
  transparent?: boolean;
  /** 单格 px */
  cellSize: number;
  /** 主线间隔（每 N 格加粗） */
  majorGridStep: number;
  /** 拼接标记颜色 */
  markerColor: string;
}
 
const DEFAULT_TILE_OPTIONS: TileOptions = {
  cellsPerPageX: 35,
  cellsPerPageY: 50,
  overlap: 2,
  background: '#FFFFFF',
  transparent: false,
  cellSize: 24,
  majorGridStep: 5,
  markerColor: 'rgba(230, 0, 18, 0.7)',
};
 
/** 单页分块结果 */
export interface TileResult {
  /** 0-based 页码（从左到右、从上到下） */
  pageIndex: number;
  /** 0-based 行号 */
  row: number;
  /** 0-based 列号 */
  col: number;
  /** 总页数 */
  totalPages: number;
  /** 该页起点在原 BeadGrid 中的 X 格坐标 */
  cellOffsetX: number;
  /** 该页起点在原 BeadGrid 中的 Y 格坐标 */
  cellOffsetY: number;
  /** 该页实际包含的格子宽度（最后一页可能小于 cellsPerPageX） */
  cellWidth: number;
  /** 该页实际包含的格子高度 */
  cellHeight: number;
}
 
/**
 * 将 BeadGrid 切成多页（A3/A4 打印）
 *
 * 拼接策略：相邻页之间有 overlap 格重叠，方便拼接对齐。
 *
 * @param grid 原始 BeadGrid
 * @param options 分页选项
 * @returns 每页的元数据（不含渲染指令）
 */
export function tileGrid(grid: BeadGrid, options: Partial<TileOptions> = {}): TileResult[] {
  const opts = { ...DEFAULT_TILE_OPTIONS, ...options };
  const stepX = Math.max(1, opts.cellsPerPageX - opts.overlap);
  const stepY = Math.max(1, opts.cellsPerPageY - opts.overlap);
 
  const cols = grid.width <= opts.cellsPerPageX ? 1 : Math.ceil((grid.width - opts.overlap) / stepX);
  const rows = grid.height <= opts.cellsPerPageY ? 1 : Math.ceil((grid.height - opts.overlap) / stepY);
  const totalPages = cols * rows;
 
  const tiles: TileResult[] = [];
  for (let row = 0; row < rows; row++) {
    for (let col = 0; col < cols; col++) {
      const x = col * stepX;
      const y = row * stepY;
      const w = Math.min(opts.cellsPerPageX, grid.width - x);
      const h = Math.min(opts.cellsPerPageY, grid.height - y);
      tiles.push({
        pageIndex: row * cols + col,
        row,
        col,
        totalPages,
        cellOffsetX: x,
        cellOffsetY: y,
        cellWidth: w,
        cellHeight: h,
      });
    }
  }
  return tiles;
}
 
/**
 * 渲染单页到 RenderCommand 列表
 *
 * 包含：背景 + 主体 + 网格线 + 拼接标记（页码 / 拼接符号）
 *
 * @param grid 原始 BeadGrid
 * @param tile 单页元数据
 * @param options 渲染选项（与 tileGrid 共享）
 * @returns RenderCommand 列表
 */
export function renderTile(
  grid: BeadGrid,
  tile: TileResult,
  options: Partial<TileOptions> = {},
): RenderCommand[] {
  const opts = { ...DEFAULT_TILE_OPTIONS, ...options };
  const cs = opts.cellSize;
  const out: RenderCommand[] = [];
  const pageW = tile.cellWidth * cs;
  const pageH = tile.cellHeight * cs;
 
  // 1) 背景（透明模式跳过）
  if (!opts.transparent) {
    out.push({ type: 'fillRect', x: 0, y: 0, w: pageW, h: pageH, color: opts.background });
  }
 
  // 2) 主体（每行合并）
  for (let y = 0; y < tile.cellHeight; y++) {
    const gridY = tile.cellOffsetY + y;
    let runStart = -1;
    let runColor = '';
    for (let x = 0; x <= tile.cellWidth; x++) {
      let curColor = '';
      if (x < tile.cellWidth) {
        const gridX = tile.cellOffsetX + x;
        const idx = grid.cells[gridY * grid.width + gridX];
        if (idx !== EMPTY_CELL) {
          curColor = grid.palette.entries[idx]?.hex ?? '';
        }
      }
      if (curColor !== runColor) {
        if (runColor && runStart >= 0) {
          out.push({
            type: 'fillRect',
            x: runStart * cs,
            y: y * cs,
            w: (x - runStart) * cs,
            h: cs,
            color: runColor,
          });
        }
        runStart = x;
        runColor = curColor;
      }
    }
  }
 
  // 3) 网格线（次线 + 主线）
  for (let i = 0; i <= tile.cellWidth; i++) {
    out.push({
      type: 'strokeLine',
      x1: i * cs,
      y1: 0,
      x2: i * cs,
      y2: pageH,
      color: 'rgba(0,0,0,0.15)',
      width: 1,
    });
  }
  for (let i = 0; i <= tile.cellHeight; i++) {
    out.push({
      type: 'strokeLine',
      x1: 0,
      y1: i * cs,
      x2: pageW,
      y2: i * cs,
      color: 'rgba(0,0,0,0.15)',
      width: 1,
    });
  }
  // 主线（每 majorGridStep 格）
  for (let i = 0; i <= tile.cellWidth; i += opts.majorGridStep) {
    if (i === 0 || i === tile.cellWidth) continue;
    out.push({
      type: 'strokeLine',
      x1: i * cs,
      y1: 0,
      x2: i * cs,
      y2: pageH,
      color: 'rgba(0,0,0,0.4)',
      width: 1,
    });
  }
  for (let i = 0; i <= tile.cellHeight; i += opts.majorGridStep) {
    if (i === 0 || i === tile.cellHeight) continue;
    out.push({
      type: 'strokeLine',
      x1: 0,
      y1: i * cs,
      x2: pageW,
      y2: i * cs,
      color: 'rgba(0,0,0,0.4)',
      width: 1,
    });
  }
 
  // 4) 拼接标记（页码 + 拼接符号）
  const labelColor = 'rgba(230, 0, 18, 0.85)';
  const labelSize = Math.max(10, Math.floor(cs * 0.4));
  // 左上角：第 X 页 / 共 Y 页
  out.push({
    type: 'fillText',
    x: 4,
    y: labelSize + 2,
    text: `第 ${tile.pageIndex + 1} / ${tile.totalPages} 页`,
    color: labelColor,
    fontSize: labelSize,
  });
  // 右上角：行号 + 列号
  out.push({
    type: 'fillText',
    x: pageW - 4,
    y: labelSize + 2,
    text: `R${tile.row + 1} C${tile.col + 1}`,
    color: labelColor,
    fontSize: labelSize,
  });
  // 拼接箭头（左侧 + 顶部）
  const arrowSize = Math.floor(cs * 0.4);
  if (tile.col > 0) {
    // 左箭头
    out.push({
      type: 'fillText',
      x: pageW / 2,
      y: -arrowSize / 2,
      text: '← 拼接',
      color: labelColor,
      fontSize: arrowSize,
    });
  }
  if (tile.row > 0) {
    // 上箭头
    out.push({
      type: 'fillText',
      x: -arrowSize * 2,
      y: pageH / 2,
      text: '↑\n拼\n接',
      color: labelColor,
      fontSize: arrowSize,
    });
  }
 
  return out;
}
